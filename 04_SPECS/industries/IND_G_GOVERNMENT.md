║  TRACK: IND-G-TYP-02                                                                                ║

║  CATEGORY: Type System Extension                                                                    ║
║  ESTIMATED EFFORT: 60-100 hours                                                                     ║
║  DEPENDENCIES: IND-G-REG-03 (NIST 800-171)                                                          ║
╠══════════════════════════════════════════════════════════════════════════════════════════════════════╣
║                                                                                                      ║
║  SCOPE:                                                                                             ║
║  Define type system for Controlled Unclassified Information handling                                ║
║  with automatic policy enforcement and marking.                                                     ║
║                                                                                                      ║
║  CUI CATEGORIES:                                                                                    ║
║  ═══════════════                                                                                    ║
║                                                                                                      ║
║  | Category        | Examples                           | RIINA Type                               ║
║  |─────────────────┼────────────────────────────────────┼─────────────────────────────────────────║
║  | CTI             | Controlled Technical Information   | CUI<T, CTI>                              ║
║  | ITAR            | Defense articles                   | CUI<T, ITAR>                             ║
║  | Export Control  | EAR-controlled data                | CUI<T, ExportControl>                    ║
║  | Privacy         | PII, PHI                           | CUI<T, Privacy>                          ║
║  | Procurement     | Source selection data              | CUI<T, Procurement>                      ║
║  | Law Enforcement | Investigative data                 | CUI<T, LawEnforcement>                   ║
║                                                                                                      ║
║  CUI TYPE SYSTEM:                                                                                   ║
║  ════════════════                                                                                   ║
║                                                                                                      ║
║  ```rust                                                                                            ║
║  // RIINA CUI Type System                                                                           ║
║                                                                                                      ║
║  /// CUI marker trait                                                                               ║
║  trait CUICategory {                                                                                ║
║      const MARKING: &'static str;                                                                   ║
║      const HANDLING_REQUIREMENT: HandlingLevel;                                                     ║
║      const DISSEMINATION_CONTROL: DisseminationControl;                                             ║
║  }                                                                                                  ║
║                                                                                                      ║
║  /// Controlled Technical Information                                                               ║
║  struct CTI;                                                                                        ║
║  impl CUICategory for CTI {                                                                         ║
║      const MARKING: &'static str = "CUI//CTI";                                                      ║
║      const HANDLING_REQUIREMENT: HandlingLevel = HandlingLevel::Basic;                              ║
║      const DISSEMINATION_CONTROL: DisseminationControl = DisseminationControl::NOFORN;              ║
║  }                                                                                                  ║
║                                                                                                      ║
║  /// ITAR-controlled defense articles                                                               ║
║  struct ITAR;                                                                                       ║
║  impl CUICategory for ITAR {                                                                        ║
║      const MARKING: &'static str = "CUI//ITAR";                                                     ║
║      const HANDLING_REQUIREMENT: HandlingLevel = HandlingLevel::Specified;                          ║
║      const DISSEMINATION_CONTROL: DisseminationControl = DisseminationControl::NOFORN_RELTO_NONE;   ║
║  }                                                                                                  ║
║                                                                                                      ║
║  /// CUI data with handling requirements                                                            ║
║  type CUI<T, Category> = Data<T>                                                                    ║
║      where Category: CUICategory                                                                    ║
║  {                                                                                                  ║
║      + MarkedWith<Category::MARKING>,                                                               ║
║      + EncryptedAtRest<FIPS140: true>,                                                              ║
║      + EncryptedInTransit<TLS: 1_2>,                                                                ║
║      + AccessControlled<authorized: CUIAuthorizedUsers>,                                            ║
║      + AuditLogged,                                                                                 ║
║      + DisseminationControlled<Category::DISSEMINATION_CONTROL>,                                    ║
║  }                                                                                                  ║
║                                                                                                      ║
║  /// Prevent CUI leakage at type level                                                              ║
║  impl<T, C: CUICategory> !Into<PublicData> for CUI<T, C> {}                                         ║
║  impl<T, C: CUICategory> !Export for CUI<T, C> where C::DISSEMINATION_CONTROL == NOFORN {}          ║
║  ```                                                                                                ║
║                                                                                                      ║
║  CUI HANDLING ENFORCEMENT:                                                                          ║
║  ═════════════════════════                                                                          ║
║                                                                                                      ║
║  ```rust                                                                                            ║

║  type CUISystem<T> = System<T>                                                                      ║
║      + NIST800171Compliant                                                                          ║
║      + FIPSCryptography                                                                             ║
║      + USPersonnelOnly                                                                              ║
║      + NoCloudStorageWithoutFedRAMP;                                                                ║
║                                                                                                      ║
║  /// CUI transmission                                                                               ║
║  type CUITransmission<T, C: CUICategory> = Transmission<CUI<T, C>>                                  ║
║      + EncryptedChannel<FIPS140: true>                                                              ║
